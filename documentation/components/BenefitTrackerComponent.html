import { Input, OnChanges } from '@angular/core';
import { Component, OnInit } from '@angular/core';
import { Benefit, Benefits } from 'src/app/models/benefits';
import { ActivatedRoute, Params } from '@angular/router';

@Component({
  selector: 'app-benefit-tracker',
  templateUrl: './benefit-tracker.component.html',
  styleUrls: ['./benefit-tracker.component.scss'],
})
export class BenefitTrackerComponent implements OnInit, OnChanges {
  id: string;
  benefit: Benefit;
  benefits: Benefit[] =  Benefits;
  benefitChosen;

  constructor(private route: ActivatedRoute) {}

  // ngOnInit() {
  //   this.id = this.route.snapshot.paramMap.get('id');
  // }

  ngOnInit() {
    this.route.params.subscribe((params: Params) => {
      this.id = params['id'];
    });
    this.benefit = this.benefits[this.id];
  }
  ngOnChanges() {
    this.route.params.subscribe((params: Params) => {
      this.id = params['id'];
    });
    this.benefits.find(benefitChosen => benefitChosen.id === +this.id);
    console.log('benefitID: ', this.benefitChosen);
    console.log('ID: ', this.id);
  }
}
