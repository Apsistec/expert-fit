  getBenefits() {
    const queryRef = this.db.collection<Benefit>('benefits', (ref) => ref.where('active', '==', true));
    return queryRef.snapshotChanges().pipe(
      map((actions) =>
        actions.map((a) => {
          const data: any = a.payload.doc.data();
          const id = a.payload.doc.id;
          return { id, ...data };
        })
      ),
      takeUntil(this.ngUnsubscribe)
    );
  }
}
